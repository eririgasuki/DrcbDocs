# 配置路由

约定报文之后，通常根据报文中的 BUSISTEP 来配置路由。

在 workspace\main-stm\static\resource\conf\tradeflowconf.json 中配置路由。

{
  "*": {
    "ReadCustIdNo_1": "main-stm/src/modules/remote-floder/ReadCustIdNo.vue",
    "ReadTOutBankCard_1": "main-stm/src/modules/remote-floder/ReadBankCard.vue"
  }
}

原理详见页面路由

配置路由之后开始编写页面，这里举例一个简化版的读取身份证页面。

在 workspace\main-stm\src\modules\remote-floder\ 中新建 ReadCustIdNo.vue 文件。

# 逻辑编写

交易页面中的逻辑大致可以分为：

  • 报文相关处理：

    • 通常在生命周期 onInitialize 中（ VUE 的created），处理已收到的报文。
    • 通常在设备成功读取介质信息后，处理要发送的报文。
    • 报文相关数据，通常放在 contextData 中。

  • 外设相关处理：

    • 根据业需，引入相应的外设模块。(引入模块禁止使用相对路径)
    • 通常在生命周期 onNavigated 中（ VUE 的mounted），开始调用外设。
    • 调用外设通常需要：
      • 语音、指示灯、文本提示。
      • 使用 try catch 捕捉设备返回的异常。
      • 捕捉到异常时需要记录日志，并弹窗提示等。
    • 通常在生命周期 onClose 中（ VUE 的destroyed），开始关闭或重置外设。
    • 报文相关数据，通常放在 deviceData 中。

  • 页面数据处理
    • 所有交易页面都需要引用并混入 BasicPage。
      • 传递了父组件的常用 props。
      • 初始化了页面倒计时。
      • 提供了常用的方法。
    • 页面相关数据，通常放在 pageData 中。
    • 页面相关逻辑开关，通常放在 trigger 中。
<script>
import BasicPage from "@main-stm/src/sdk/mixins/basic-page-noflow";
import { IDCardReader } from "awp-plugin-device-idcard-reader";
import { SensorsIndicators } from "awp-plugin-device-sensors-indicators";
import { AudioPlayer } from "awp-lib-audio-player";
import { TradeLogger } from "awp-plugin-logger";
import { ExceptionHelper } from "@main-stm/src/sdk/utils/exception-helper";
import { Moment } from "awp-lib-moment";

export default {
  name: "read-cust-id-no",
  mixins: [BasicPage],
  data() {
    return {
      pageData: {
        title: "",
      },
      contextData: {},
      deviceData: {},
      trigger:{
        canSubmit: false,
        needReset: false
      }
    };
  },
  methods: {
    onInitialize() {
      this.initPage();
    },

    async onNavigated() {
      await this.readIDCardAsync();
      await this.checkCardDataAsync();
      this.trigger.canSubmit && await this.submit();
    },

    async onClose() {
      SensorsIndicators.closeIDCardGuidLightAsync();
      try {
        await IDCardReader.cancelAsync();
      } catch (error) {
        console.log("error", error);
      }
      this.trigger.needReset && await this.resetIDCardAsync();
    },


    initPage() {
      this.contextData.agentFlag = this.find("AGENTFLAG");
      //非本人办理时，页面提示语不同
      let agentMap = {
        ReadCustIdNo_1: "账户本人",
        ReadAgentIdNo_1: "代理人",
        ReadGuardIdNo_1: "监护人"
      };
      let whose = this.contextData.agentFlag === "0" ? "您的" : agentMap[this.BUSISTEP];
      this.pageData.title = `请插入${whose}身份证`;
    },


    async readIDCardAsync() {
      AudioPlayer.play("insertIDCard");
      SensorsIndicators.openIDCardGuidLightAsync();
      try {
        this.deviceData.IDData = await IDCardReader.readIdCardAsync();
      } catch (error) {
        this.trigger.needReset = true;
        console.error("设备读取身份证失败:", error);
        TradeLogger.error("设备读取身份证失败:", error);
        ExceptionHelper.pushError("设备读取身份证失败");
      }
      console.log("身份证数据", this.deviceData.IDData);
    },


    async resetIDCardAsync() {
      try {
        await IDCardReader.resetAsync(1);
      } catch (error) {
        console.error("身份证模块重置异常:", error);
        TradeLogger.error("身份证模块重置异常:", error);
      }
    },


    async checkCardDataAsync() {
      if (this.isOverdue(this.deviceData.IDData.EndDate)) {
        TradeLogger.error("身份证过期不能办理");
        ExceptionHelper.pushError("身份证过期不能办理");
        return false;
      }
      this.trigger.canSubmit = true;
    },


    isOverdue(endDate) {
      let now = Moment.format("yyyyMMdd");
      return endDate - now < 0;
    },


    submit() {
      let info = {
        IDTYPE: this.contextData.IDType,
        IDNO: this.deviceData.IDData.ID,
        IDNAME: this.deviceData.IDData.Name,
        IDADDRESS: this.deviceData.IDData.Address,
        IDBIRTHDATE: this.deviceData.IDData.Address,
        ISSUEOFFICE: this.deviceData.IDData.Birthday,
        IDSEX: this.deviceData.IDData.Sex === "女" ? "2" : "1",
        IDCERTDUEDATE: this.deviceData.IDData.EndDate
      };
      this.submitAsync(info);
    }
  }
};
</script>
