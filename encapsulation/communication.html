# 简述

  • 在 main.js 中已创建浏览器与 Cordova 服务的 websocket 连接进行通讯。
  • 交易页面中，通常直接使用封装好请求头之后的方法发送 HTTP 请求：
    1. 使用 submit 方法发送交易数据给 P 端。
    2. 使用 sendMessage 方法发送交易状态给 A 端。
    3. 使用 RequestHelper.sendAsync 方法调用数据校验等接口。

# 前端通信桥接器 CommunicationClient

# 请求处理 RequestHelper

  • RequestHelper 是对前端通信桥接器 CommunicationClient 的一层封装，确立了请求头格式，服务器地址等内容。
  • 通过 Ameba.session 上的数据赋值给请求头，传入的 content 参数作为请求体。
  • 服务码 TransServiceCode 即发送数据或校验数据的接口名。
  • 封装请求 request = {SYS_HEAD: sysHead, APP_HEAD: appHead, REQ_BODY: content} 后通过 CommunicationClient 发送报文 。
import SysHead from '@main-stm/src/messages/ayo/sys-head';
import AppHead from '@main-stm/src/messages/ayo/app-head';
import {  Moment  } from 'awp-lib-moment';
import {  CommunicationBridge, CommunicationClient  } from 'awp-plugin-communication-bridge';
import {  Ameba  } from 'awp-lib-ameba-js';

let sessionClient;
let isPauseServiceForException = true;
let ServerIP = 'http://10.8.221.253:6661';  //用于更改为云柜台的服务器地址
export default {
  async sendAsync(content, ms = 500) {
    let transServiceCode = content.TransServiceCode; // 服务码
    if (!sessionClient) {
      let session = Ameba.session;
      sessionClient = session.Client;
      let result = await this.communicationSendAsync(content, sessionClient, transServiceCode, ms);
      return result;
    } else {
      let result2 = await this.communicationSendAsync(content, sessionClient, transServiceCode, ms);
      return result2;
    }
  },
  async communicationSendAsync(content, sessionClient, transServiceCode, ms) {
    let sysHead = new SysHead();
    let appHead = new AppHead();

    // 设置系统报文头
    this.setSysHead(sysHead, sessionClient);
    // 设置服务码
    sysHead.TransServiceCode = transServiceCode;
    // 设置服务场景码
    sysHead.TransServiceScene = content.TransServiceScene;
    // 设置应用报文头
    this.setAppHead(appHead, Ameba.session);
    let request = {
      SYS_HEAD: sysHead,
      APP_HEAD: appHead,
      REQ_BODY: content
    };
    console.log("<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<上送报文>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>");
    console.log(request);
    TradeLogger.info("上送报文:", request);
    let startTime = new Date().getTime();
    try {
      let CommunicationBridgeToAyo = new CommunicationClient("httpAYO");
      let result = await CommunicationBridgeToAyo.sendAsync(request, isPauseServiceForException);
      console.log("<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<<接受报文>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>");
      console.log(result);
      TradeLogger.info("接受报文:", result);
      let endTime = new Date().getTime();
      let timeSpan = endTime - startTime;
      if (timeSpan < ms) {
        await delayAsync(ms - timeSpan);
        return result;
      } else {
        return result;
      }
    } catch (error) {
      console.error(error);
      TradeLogger.info("通讯异常:", error);
      throw new Error(error);
    }
  },
  /**
   * 设置系统报文头
   * @param {*} sysHead
   * @param {*} sessionClient
   */
  setSysHead(sysHead, sessionClient) {
    sysHead.ConsumerIP = sessionClient.SysHead.ConsumerIP;
    sysHead.ServerIP = ServerIP;//sessionClient.SysHead.ServerIP;  //用于连接云柜台的服务器
    sysHead.MacValue = sessionClient.SysHead.MacValue;
    sysHead.DvcNum = sessionClient.SysHead.DvcNum;
    sysHead.ConsumerId = "C001";//sessionClient.SysHead.ConsumerId;  //使用云柜台的渠道号
  },
  /**
   * 设置报文apphead
   * @param {*} appHead
   * @param {*} sessionClient
   */
  setAppHead(appHead, sessionClient) {
    appHead.FILEINDXNO = sessionClient.Customer.FileIndexNO ? sessionClient.Customer.FileIndexNO : "";
    appHead.TdgBrah = sessionClient.Client.DeviceInfo.TdgBrah;
    appHead.DeviceType = sessionClient.Client.DeviceInfo.DeviceType;
    appHead.DeviceNum = sessionClient.Client.DeviceInfo.DeviceNum;
    appHead.TranTellerNo = sessionClient.Client.DeviceInfo.TranTellerNo;
    appHead.TASKID = sessionClient.Trade.TaskID; 
    appHead.TdgBrahName = sessionClient.Client.DeviceInfo.TdgBrahName; 
  },
};

# 混入 BasicPage

  • BasicPage 封装了云柜台交易中最常用到的服务码 TransServiceCode 为 submittask 的数据提交等接口。
  • 数据提交接口中所需的 FLOWID, TASKID, NODEID 等请求字段在交易上下文 context 中可以获取。
  • 交易页面中获取的数据通存作为 info 传入封装的 submit 方法里。
// 导入流程引擎
import Submittask from "@main-stm/src/messages/ayo/submittask";
import PushMessage from '@main-stm/src/messages/ayo/pushMessage';
import RequestHelper from "@main-stm/src/sdk/utils/request-helper-ayo";

export default {
    /**
     * 业务节点提交
     * @param {*} context
     * @param {*} info
     */
    submit(context, info) {
      let request = new Submittask();
      request.FLOWID = context.REQ_BODY.FLOWID;
      request.TASKID = context.REQ_BODY.TASKID;
      request.NODEID = context.REQ_BODY.NODEID;
      request.NODENAME = context.REQ_BODY.NODENAME;
      request.BUSISTEP = context.REQ_BODY.BUSISTEP;
      request.FLOWNAME = context.REQ_BODY.FLOWNAME;
      request.MMDATE = context.APP_HEAD.MMDATE;
      request.MMID = context.APP_HEAD.MMID;
      request.FACEDATAIN = info;
      return new Promise(function (resolve, reject) {
        RequestHelper.sendAsync(request).then(res => {
          resolve(res);
        }).catch(err => {
          reject(err);
        });
      });
    },
  }
};

# 校验密码页面 CheckPinkeyPad.vue

  • CheckPinkeyPad 页面获取用户输入的密码 pinBlock 后，通过点击确定按钮，触发 commitAsync 方法提交数据。
    • 使用 BasicPage 混入页面的 submit 方法提交数据。
    • 使用引入的 RequestHelper 调用服务码 TransServiceCode 为 TC015 的密码校验接口进行校验。
<template>
  <page>
    <page-header :hasUnderline="hasUnderline">{{title}}</page-header>
    <page-content justify-content="center">
      <aui-box direction="column" justify-content="center" align-items="flex-start">
        <label class="label">请输入密码</label>
        <pin-keypad @passwordgotton="GetPassword" />
      </aui-box>
      <img src="static/resource/img/input-password.png" alt />
    </page-content>
    <page-footer>
      <aui-button cornerType="rounded-corner" @click="commitAsync" :disabled="isDisabled">确定</aui-button>
    </page-footer>
  </page>
</template>

<script>
import BasicPage from "@main-stm/src/sdk/mixins/basic-page-noflow";
import PinKeypad from "@main-stm/src/components/PinKeypad";
import RequestHelper from "@main-stm/src/sdk/utils/request-helper-ayo";
import TC015 from "@main-stm/src/messages/ayo/tc015"; //密码检查

export default {
  name: "check-pinkey-pad",
  mixins: [BasicPage],
  data() {
    return {
      title: "请输入您的账户密码",
      hasUnderline: false,
      pinBlock: "",
      isDisabled: true,
    };
  },
  methods: {
    GetPassword(pinBlock) {
      this.pinBlock = pinBlock;
      if (this.pinBlock != "") {
        this.isDisabled = false;
      } else {
        return;
      }
    },
    //向p端推送密文
    async commitAsync() {
      let info = [
        {
          FIELDCODE: "CHKPINBLOCK",
          FIELDVALUE: this.pinBlock
        }
      ];
      let response = "";
      try {
        response = await this.submit(this.context, info);
      } catch (error) {
        console.error(error);
      }
      this.checkPSW();
    },
    //调用TC015接口校验密码
    async checkPSW() {
      let request = new TC015();
      request.CHKPINBLOCK = this.pinBlock;
      let result = "";
      try {
        result = await RequestHelper.sendAsync(request);
      } catch (error) {
        console.error(error);
      }
      //返回code值为000000表示成功
      if (result.SYS_HEAD.ReturnCode == "000000") {
        console.debug("密码正确");
      } else {
        console.error(result);
        TradeLogger.error("失败:", result);
      }
    }
  },
  components: {
    PinKeypad
  }
};
</script>
