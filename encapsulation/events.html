  • Cordova 服务在触发事件后回调 JS SDK 进行处理。

# 远程指令管理 RemoteCommandManager

  • RemoteCommandManager 是对远程指令类型的事件归类封装的 EventManager 。
import { EventManager } from 'awp-plugin-event';
const eventManager = new EventManager();
export default {
  addHandler(commandType, handler) {
    eventManager.addEventListener(commandType, handler);
  },
  removeHandlers(commandType) {
    eventManager.removeEventListeners(commandType);
  },
  invokeCommand(commandType, commandParameters) {
    eventManager.raiseEvent(commandType, commandParameters);
  }
};

# 事件交互插件 EventHubInterop

  • 在 app-loader.js 中引入并安装 EventHubInterop 事件交互插件。
  • 当 websocket 返回 Cordova 服务触发的类型为 RemoteCommandReceived || ReverseCommandReceived 事件时，执行相应函数。
import RemoteCommandManager from './remote-command-manager';

export default {
  install() {
    this.initialize();
  },

  initialize() {
    cordova.exec(eventData => this.handleEvent(eventData),
      error => { throw error; }, 'EventHubInterop', 'Initialize');
  },

  handleEvent(eventData) {
    if (eventData.EventType === 'RemoteCommandReceived') {
      RemoteCommandManager.invokeCommand(eventData.EventParameters.CommandType,
        eventData.EventParameters.CommandParameters);
    } else if (eventData.EventType === 'ReverseCommandReceived') {
      if (navigator.userAgent.indexOf('Crosswalk') > -1) {
        RemoteCommandManager.invokeCommand(eventData.EventParameters.CommandType,
          JSON.parse(eventData.EventParameters.CommandParameters));
      } else {
        RemoteCommandManager.invokeCommand(eventData.EventParameters.CommandType,
          eventData.EventParameters.CommandParameters);
      }
    } 
  }
};

# 云柜台导航页面 PageContainerNavigator.vue

  • 在 PageContainerNavigator 中，就是通过监听 openPage 事件的消息来实现交易页面的切换。执行函数内容详见 Router 。
<template>
  <div class="page-container-navigator" id="remotepagecontainernavigator">
    <transition name='page' mode="out-in" @after-enter='onNavigated'>
        <component ref="currentPage" class="page trade-content-font" :key="indexDate" :is="currentPage" :BUSISTEP="BUSISTEP" :session="session" :context="context" @openPage="openPage" @exit="exitTrade" @abort="abortTrade"></component>
    </transition>
  </div>
</template>

<script>
import RemoteCommandManager from "@main-stm/src/sdk/remote-command-manager";
import RemoteCommandTypes from "@main-stm/src/enums/remote-command-types";
export default {
  name: "page-container-navigator",
  methods: {
    async onInitialize() {
      //注册remote事件
      RemoteCommandManager.addHandler(
        RemoteCommandTypes.openPage,
        this.onOpenPage.bind(this)
      );
      RemoteCommandManager.addHandler(
        RemoteCommandTypes.exit,
        this.onExit.bind(this)
      );
    },
    onClose() {
      //注销remote事件
      RemoteCommandManager.removeHandlers(RemoteCommandTypes.openPage);
      RemoteCommandManager.removeHandlers(RemoteCommandTypes.exit);
    },
};
</script>
