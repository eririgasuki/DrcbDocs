# 简述

  • 页面分为两层，其中 Activity 为首，通过 ActivityManager 封装的 Router 实现 Activity 中页面切换。
  • 随后通过 Activity 之一的 PageContainerNavigator 作为父组件，使用动态组件 <component/> 加载子组件的方式，来实现云柜台交易页面切换。

# Activity 视图管理器 RootActivityManager

  • 在 app-loader.js 中 RootActivityManager.router = router 已将 router 挂载到 RootActivityManager 对象的 router 上。

  • 在 App.vue 中 RootActivityManager.changeActivityManager(new PortalActivityManager()) 执行 changeActivityManager 方法时：

    1. 将 router 挂载到 PortalActivityManager 上便于使用 replace 方法切换页面。
    2. 将 PortalActivityManager 挂载到 currentActivityManager 上。
    3. 执行 initialize 方法。
/**
 * @module 【ui】RootActivityManager
 * @desc Activity视图的管理器
 * （主要是控制和切换不同活动视图的切换）
 */
var _default = {
  router: undefined,
  currentActivityManager: undefined,
  /**
   * @func changeActivityManager
   * @desc 切换视图
   * @param {Object} activityManager - activity管理对象
   * @example import { RootActivityManager } from 'awp-ui-activity-manager'
   *
   * RootActivityManager.changeActivityManager(activityManager)
   */
  changeActivityManager: function changeActivityManager(activityManager) {
    activityManager.router = this.router;
    if (this.currentActivityManager === activityManager) {
      return;
    }
    if (this.currentActivityManager && this.currentActivityManager.close) {
      this.currentActivityManager.close();
    }
    this.currentActivityManager = activityManager;
    if (activityManager) {
      activityManager.initialize();
    }
  }
};
_exports.default = _default;

# Activity 入口视图管理器 PortalActivityManager

  • 执行 initialize 方法，将路由切换 this.router.replace('/ads') 到广告页面。
  • 之后通过绑定上 Vue 的 this.$activityManager 执行 requestUpdateView 方法切换 Activity 页面。
import EventName from '@main-stm/src/activity-managers/activity-event-names';
export default class PortalActivityManager {
  initialize() {
    this.gotoAds();
  }
  gotoAds() {
    this.router.replace('/ads');
  }
  async requestUpdateView(eventName, eventArgs) {
    if (eventName === EventName.PAUSE_COMMAND_RECEIVED) {
      this.router.replace("/out-of-service");
    }else if (eventName === EventName.REMOTE_VIDEO_CLIENT) {
      this.router.replace('/page-container-navigator'); 
    } else if (eventName === EventName.SERVICE_TYPE_CLIENT) {
      this.router.replace('/service-type-menu'); 
    }
  }
}

# Activity 广告页面 Ads.vue

  • 通过 mixins: [BasicActivity] 将 RootActivityManager.currentActivityManager 挂载到 this.$activityManager 上。
  • 通过点击页面触发 onClicked 方法，执行 requestUpdateView(EventNames.SERVICE_TYPE_CLIENT) 将路由切换到 this.router.replace('/service-type-menu') 服务类型选择页面。
<template>
  <div class="ads" @click="onClicked">
    <ads-slide></ads-slide>
  </div>
</template>

<script>
import AdsSlide from "@main-stm/src/components/AdsSlide";
import BasicActivity from "@main-stm/src/sdk/mixins/basic-activity";
import EventNames from "@main-stm/src/activity-managers/activity-event-names";
export default {
  name: "ads",
  mixins: [BasicActivity],
  components: {
    AdsSlide
  },
  methods: {
    onClicked() {
      this.$activityManager.requestUpdateView(EventNames.SERVICE_TYPE_CLIENT); //服务类型选择
    }
  }
};
</script>

# Activity 云柜台导航页面 PageContainerNavigator.vue

  • 通过动态组件 <component/> 实现云柜台交易页面的跳转。
  • 动态组件绑定 :is="currentPage" ,根据 this.currentPage = await this.getPage(PagePath) 动态加载组件。
  • 通过 getPage 方法传入组件路径 PagePath 加载页面。
  • 在初始化时，设置 const PagePath = "main-stm/src/modules/remote-floder/VideoConnect.vue" 加载视频连接页面。
  • 初始化时，监听远程指令消息，通过消息 pageInfo 中的 BUSISTEP 或 PAGETYPE 找到 TradeFlower 中配置的对应组件路径，加载相应交易页面。
<template>
  <div class="page-container-navigator" id="remotepagecontainernavigator">
    <view-header :tradetitle="title" :tradeType="tradeType"></view-header>
    <div class="trade-page-container">   
      <transition name='page' mode="out-in" @after-enter='onNavigated'>
          <component ref="currentPage" class="page trade-content-font" :key="indexDate" :is="currentPage" :BUSISTEP="BUSISTEP"  :session="session" :context="context" @openPage="openPage" @exit="exitTrade" @abort="abortTrade"></component>
      </transition>
    </div>
    <aui-row id="footer">
      <aui-col span="24" class="tip">注意保护个人信息，请勿向他人泄露，谨防受骗！</aui-col>
    </aui-row>
  </div>
</template>

<script>
import ViewHeader from "@main-stm/src/components/ViewHeader";
import RemoteCommandManager from "@main-stm/src/sdk/remote-command-manager";
import RemoteCommandTypes from "@main-stm/src/enums/remote-command-types";
import TradeFlower from "@main-stm/static/resource/conf/tradeflowconf.json";
export default {
  name: "page-container-navigator",
  data() {
    return {
      currentPage: undefined,
      BUSISTEP: ""  //记录当前跳转的是哪个页面，实现公用一个页面，区分提交所送字段
    };
  },
  methods: {
    /**
     * 加载页面函数
     * path:要跳转的页面路径
     * page:返回要加载的页面
     */
    getPage(path) {
      return new Promise(async (resolve, reject) => {
        await require([path], page => {
          resolve(page.default);
        });
      });
    },
    async onInitialize() {
      RemoteCommandManager.addHandler(
        RemoteCommandTypes.openPage,
        this.onOpenPage.bind(this)
      );
      const PagePath = "main-stm/src/modules/remote-floder/VideoConnect.vue";
      this.currentPage = await this.getPage(PagePath);
    },
    /**
     * @description 打开单页面
     */
    async onOpenPage(pageInfo) {
      let pagePath;
     if( TradeFlower["*"][pageInfo.REQ_BODY.PAGETYPE]){
        pagePath = TradeFlower["*"][pageInfo.REQ_BODY.PAGETYPE];
      }
      else {
        pagePath = TradeFlower[pageInfo.REQ_BODY.PAGENO][pageInfo.REQ_BODY.BUSISTEP];
      }
      if (pagePath) {
        this.currentPage =await this.getPage(pagePath);
      } else {
        console.error("没有配置节点路径");
      }
    },
  components: {
    ViewHeader
  }
};
</script>
