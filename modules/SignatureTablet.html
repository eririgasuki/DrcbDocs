行方业务需要应用记录的电子流水日志，在一些交易中的部分场景，会显得比较突兀。 例如转账时，日志中用户插入介质签完名就交易结束了，不知道用户做了什么。因为转账等操作是在坐席端完成。

后续应用可能会按行方需求，在签名时记录一些重要交易信息。

async getVoucherLog() {
  let elecData = this.find("ELECDATA");
  let flowID = this.find("FLOWID");
  if (flowID == "RuralCreditBankTC") {
    this.writeLogFlowAsync(`付款方式：${elecData.MEDIATYPE}`);
    this.writeLogFlowAsync(`交易金额：${elecData.ACTAMOUNT}`);
    if (elecData.FEEAMOUNT) {
      this.writeLogFlowAsync(`手续费收取方式：${elecData.TAKEFEEFLAG}`);
      this.writeLogFlowAsync(`手续费金额：${elecData.FEEAMOUNT}`);
    }
  }
  if (flowID == "RuralCreditBankTD") {
    this.writeLogFlowAsync(`资金去向：${elecData.AmtTarget}`);
    this.writeLogFlowAsync(`交易金额：${elecData.ACTAMOUNT}`);
    if (elecData.FEEAMOUNT) {
      this.writeLogFlowAsync(
        `手续费收取方式：${
          elecData.JG_FLAG == 1 ? "柜台收取" : elecData.FEEFLAG
        }`
      );
      this.writeLogFlowAsync(`手续费金额：${elecData.FEEAMOUNT}`);
    }
  }
},

当A端推送MIDSTATUEFLAG字段值为1时，显示交易异常而非交易成功。

this.find("ELECDATA").MIDSTATUEFLAG == "1" &&
  (this.title = "交易异常") &&
  (this.titleImg = "static/resource/trade/AuthorizationFailer.png");

当交易场景为交割时，通常由柜面生成电子凭证，云柜台不生成。

this.tranCreateElecFlg = this.context.REQ_BODY.SHOWDATA.find(r => {
  return r.FIELDCODE == "TranCreateElecFlg"; //是否生成电子凭证  1-生成，0-不生成
}).FIELDVALUE;

由于需求不断变更的原因，目前设计的电子凭证模板，签字图片，交易结果，核证行是分开的。 电子凭证生成页面中，电子凭证作为模板字符串插入完整的（包含签字，交易结果，核证行）html模板。 部分交易含有交易结果的，需根据交易新增对应的完整html模板。

async CreatePdf() {
  if (
    this.context.REQ_BODY.FLOWID == "PersOpenAcct" &&
    this.PrintData.OPENTYPE == "2"
  ) {
    this.workPathEnd =
      this.workPath +
      "\\workspace\\main-stm\\static\\resource\\template\\ygt\\PBVoucherTemplateHtml.html";
  } else {
    this.workPathEnd =
      this.workPath +
      "\\workspace\\main-stm\\static\\resource\\template\\ygt\\VoucherTemplateHtml.html";
  }
  let printhtml = await FileReader.readAllTextAsync(this.workPathEnd); //读取交易对应的模板
  let xtemplate4 = new XTemplate(printhtml); //新建模板的XTemplate
  this.printStr = xtemplate4.render({ printData: this.PrintData }) //渲染模板
  await HTMLToPDF.HtmlToPdf(this.printStr, this.voucherPDFFilePath); //生成电子凭证
},
