# AuthDialog 指纹授权弹窗组件

# 简易流程图

# 注意点&可能的扩展改造

指纹校验接口TC026需要设备类型DEVICE_NO参数，是通过获取状态的结果映射的，以后可能扩展。

status = await FingerprintReader.getStatusAsync();
vendorMap = {
  "1": "T0", //天诚
  "2": "Z0", //中正
  "3": "W0"  //维尔
}
deviceNo = vendorMap[status.Extra.FingerVendor]

最初指纹弹窗有确认和取消两个按钮，按业务要求，功能全部整合到一个确定按钮里：

  • 没有按指纹的时候，点确定不能取消弹窗:
onCancel() {
  if (!this.fingerPrintVal) {
    this.tipMessage = "请按下指纹";
    return;
  }
  this.result.success = false;
  Dialog.close(this);
},
  • 获取指纹失败或校验失败后，点确定才能退出弹窗：
try {
  this.fingerPrintVal = await FingerprintReader.getFeatureDataAsync();
} catch (error) {
  this.tipMessage = "获取指纹失败！请重新授权";
  this.fingerPrintVal = "errorButCanCancel"; //避免获取不到指纹时无法关闭弹窗
}

指纹弹窗按钮入参带有柜员号时，会要求授权柜员与该柜员号（通常是接单柜员）一致。 且部分交易（柜员收取客户现金场景）指纹校验失败5次后，需要结束交易。

//调用弹窗的页面
let result = await Dialog.showAwait(AuthDialog, {
  dialogBoxContentArgs: {
    timeoutSeconds: 120,
    authTellerNo: this.PADNo,
    endTradeAsync: async () => {
      await this.endPadTask();
      this.pushAsync(
        {
          ERRORCODE: "999999",
          ERRORMSG: "指纹验证失败5次，终止交易"
        },
        true,
        "0004",
        "",
        "CashAuthException"
      );
    }
  }
});

//AuthDialog
watch: {
  authFailedTimes(curVal) {
    if (curVal >= 5 && this.dialogBoxContentArgs.endTradeAsync) {
      Dialog.close(this);
      ExceptionHelper.pushError("指纹验证失败5次，终止交易");
      this.dialogBoxContentArgs.endTradeAsync();
    }
  }
},
async authorizationCheckAsync() {
  let request = new TC026(); 
  request.DEVICE_NO = this.deviceNo;
  request.FINGERINFO = this.fingerPrintVal;
  let result = await RequestHelper.sendAsync(request);
  if (result.SYS_HEAD.ReturnCode === "000000") {
    if (this.authTellerNo) {
      //指纹验证参数如果传了柜员号就需要验证接单柜员和指纹验证柜员一致
      if (result.RSP_BODY.AUTH_USER_SSO_USER_ID === this.authTellerNo) {
        this.result.success = true;
        Dialog.close(this);
      } else {
        this.result.success = false;
        this.tipMessage = "非接单柜员授权";
        this.authFailedTimes++;
      }
    } else {
      this.result.success = true;
      this.result.tellerNo = result.RSP_BODY.AUTH_USER_SSO_USER_ID;
      Dialog.close(this);
    }
  } else {
    this.result.success = false;
    this.authFailedTimes++;
  }
}
