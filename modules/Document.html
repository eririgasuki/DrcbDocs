对账单A4纸一页最多打60行，且要素较多，为了避免换行已经尽力权衡调整每列宽度。 最终摘要（目前最多12个汉字加一个符号）和对方户名（最多30个汉字）可能会影响行数。

dividePages() {
  //满60行算一页
  let currentLines = 0; //当前行数
  let lastLines = 0; //记录遍历到某条数据开始时，已有行数。用于计算长度换行后，行数超过60的场景
  let divider = 0; //分页时的数据索引下标
  for (let i = 0; i < this.List.length; i++) { //遍历整个数据List
    lastLines = currentLines;
    currentLines++; //每遍历一条数据，当前行数+1
    //内容太长则换行
    if (this.hasCounter) { //含对方户名时，摘要每5个字、对方户名每20个字时，当前行数+1
      (this.List[i].COUNTER_ACCT_NAME.length > 20 || 
        (Summary[this.List[i].REMARK_CODE] &&
          Summary[this.List[i].REMARK_CODE].length > 5)) &&
        currentLines++;
      (this.List[i].COUNTER_ACCT_NAME.length > 40 ||
        (Summary[this.List[i].REMARK_CODE] &&
          Summary[this.List[i].REMARK_CODE].length > 10)) &&
        currentLines++;
    } else { //不含对方户名时，摘要每7个字时，当前行数+1
      Summary[this.List[i].REMARK_CODE] &&
        Summary[this.List[i].REMARK_CODE].length > 7 &&
        currentLines++;
    }
    //行数只能小于等于60
    if (currentLines > 60) { //如果这条数据导致行数超过60行了，会将这条数据插到下一页。
      this.printList.push(this.List.slice(divider, i));
      divider = i;
      currentLines = currentLines - lastLines; //下一页的起始行数为将这条数据的行数。
    } else if (currentLines === 60 || i === this.List.length - 1) {
      this.printList.push(this.List.slice(divider, i + 1));
      divider = i + 1;
      currentLines = 0;
    }
  }
},

利用Vue实例的emit和on方法将打印机抛出的事件信息，传递到打印进度的弹出中。

//cunrrentPrinter中
const vm = new Vue();
DocumentPrinterEventManager.addOnMediaPresentListener(
  this.$options.name,
  args => { //触发MediaPresent事件时执行的函数
    vm.$emit("communication", args);
  }
);
Dialog.showNonAwait(PrintTipSpec, {
  dialogBoxContentArgs: {
    vm,
  }
});

//PrintTipSpec弹窗中
mounted() {
  this.dialogBoxContentArgs.vm.$on('communication', args => {
      this.tips = "正在打印第: " + args.WadIndex + "张, 总共" + args.TotalWads + "张";
      this.$refs.loadBarInner.style.width = (args.WadIndex/args.TotalWads)*100 + '%';
  });
},
