# 入口文件 main.js

  • 引入 Jsbridge ：创建一个 websocket 连接，让浏览器通过 WebSocket 与 Cordova 服务进行通讯。
  • 引入 NativeEventManager ：将 Native 层事件的发布订阅中心挂载在 window.AgreeSDK 上。
  • 引入 app-loader ,开始加载创建 Vue 应用。
import { Jsbridge } from 'awp-lib-jsbridge';
import { NativeEventManager } from 'awp-plugin-event';

let jsbridge = new Jsbridge({
  url: 'ws://127.0.0.1:53005/bridge'
});

// 兼容旧代码
window.cordova = jsbridge;

window.AgreeSDK = {};
window.AgreeSDK.nativeEventManager = NativeEventManager;
window.AgreeSDK.jsbridge = jsbridge;

console.log("==========!!!!!!============");
require(['@main-stm/src/app-loader']);

# 应用加载 app-loader.js

  • 引入 Vue 的路由，状态管理核心插件。
  • 引入 app-init 开始初始化 Vue 使用的插件和全局注册组件。
  • 引入 RootActivityManager : 将 router 挂载到 RootActivityManager.router 上。
  • 引入 Ameba : 读取 sdb(SessionDataBasket) 目录下的共享数据篮子文件。
  • 引入 SessionHelper : 获取 SessionData ,并将获取到的数据赋值到 Ameba.session 上。
import Vue from 'vue';
import VueRouter from 'vue-router';

import App from './App';
import routes from './routes';
import store from './vuex/store';

import { Ameba } from 'awp-lib-ameba-js';
import { SessionHelper } from 'awp-plugin-session-helper';
import { RootActivityManager } from 'awp-ui-activity-manager'; // 路由管理

require(['@main-stm/src/app-init'], async () => {
  const router = new VueRouter({
    routes,
    history: false
  });
  RootActivityManager.router = router; //路由管理

  try {
    await Ameba.initializeAsync({
      sdbUrlList: ["/main-stm/src/sdb/Client.sdb", "/main-stm/src/sdb/Customer.sdb", "/main-stm/src/sdb/Trade.sdb", "/main-stm/src/sdb/Teller.sdb","/main-stm/src/sdb/TransData.sdb"],
      wwwRoot: '',
      dynamicModuleType: 'AMD' // 'CommonJs' 或者 'AMD'
    });
  } catch (err) {
    console.log(err);
  }

  const session = await SessionHelper.getSessionDataAsync();
  if (Ameba.session.Client) {
    const clientFieldNames = Object.keys(Ameba.session.Client);
    Object.keys(session.Client).forEach(key => {
      if (clientFieldNames.includes(key)) {
        Ameba.session.Client[key] = session.Client[key];
      }
    });
  }

  console.log('session ===>', session);

  /* eslint-disable no-new */
  new Vue({
    el: '#app',
    store,
    router: router,
    render: h => h(App)
  });
});

# 应用初始化 app-init.js

  • 引入 Vue 的路由，状态管理核心插件。

  • 引入 VueResource : AJAX插件，Vue2.0后不再维护，如今取而代之的是 axios。

  • 引入 AUISS : 赞同UI框架。

  • 引入 OutFocus, FocusManager : 焦点控制插件。

  • 引入 VueDispatcher : 为焦点控制插件设置函数执行的时机为 nextTick。

  • 引入 GlobalTimer : 全局倒计时插件。

  • 引入 EventHubInterop : 事件交互中心，处理远程和本地事件通知。

  • 引入 PageComponentsLoader : 将页面划分为 <page-group/>,<page/>,<page-header/>,<page-content/>,<page-footer/> 并全局注册这些组件。

  • 引入 FormItem, TableOperation, InputErrorTemplate : 全局注册表单验证，加钞提示弹窗表格，输入验证组件。

  • 引入 UncaughtErrorHandlerInitializer : 错误处理插件。

  • 引入 AudioPlayer : 音频播放插件。

# 应用页面 App.vue

  • 引入 RootActivityManager, PortalActivityManager : 在创建 Vue 应用时初始化 Activity 视图的管理器，切换 PortalActivityManager 设置的页面为入口页面。
  • <router-view/> 切换时，如果存在 onNavigated 生命周期钩子函数就执行该函数。
<template>
  <div id="app">
    <transition name="activity-fade" mode="out-in" @after-enter="runTrade">
      <router-view ref="routerView"></router-view>
    </transition>
  </div>
</template>

<script>
  import { RootActivityManager } from "awp-ui-activity-manager";
  import PortalActivityManager from "@main-stm/src/activity-managers/portal-activity-manager";
  export default {
    name: "app",
    created() {
      RootActivityManager.changeActivityManager(new PortalActivityManager());
    },
    methods: {
      runTrade() {
        if (this.$refs.routerView.onNavigated) {
          this.$refs.routerView.onNavigated();
        }
      }
    }
  };
</script>
